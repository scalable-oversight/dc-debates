import torch, time, math

def prefix_sum(x: torch.Tensor) -> torch.Tensor:
                                                                     
    is_pos = (x > 0).to(torch.int32)
                                 
    pos_cnt = is_pos.cumsum(0, dtype=torch.int32)
                                                             
    prev_parity = (pos_cnt - is_pos) & 1             
                                                         
    masked = x * prev_parity.to(x.dtype)
    return masked.cumsum(0, dtype=x.dtype)
